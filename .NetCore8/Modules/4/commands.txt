    options.Cookie.Path = "/";
    options.Cookie.Name = "__Host-Session";

 private void DeleteAllCookies()
        {
            foreach (var cookie in Request.Cookies.Keys)
            {
                Response.Cookies.Delete(cookie, new CookieOptions()
                {
                    Secure = true,
                });
            }
        }
